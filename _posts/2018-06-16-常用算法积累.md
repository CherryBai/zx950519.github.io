## 二分搜索

```
  public static boolean binarySearchByLoop(int x, int[] data) {
        int low = 0;
        int high = data.length-1;
        while (low<=high) {
            int middle = (low + high)/2;
            if(x == data[middle]) {
                return true;
            }
            else if(data[middle] < x) {
                low = middle + 1;
            }
            else {
                high = middle - 1;
            }
        }
        return false;
    }

    public static boolean binarySearchByRecursion(int key, int low, int high, int[] data) {
        int middle = (high - low) / 2 + low;
        if(low>high) {
            return false;
        }
        if(data[middle]==key)
            return true;
        else if(key > data[middle])
            return binarySearchByRecursion(key, middle+1, high, data);
        else if(key < data[middle])
            return binarySearchByRecursion(key, low, middle-1, data);
        else
            return false;
    }
```

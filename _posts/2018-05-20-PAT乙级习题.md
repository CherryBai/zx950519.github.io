    public static long getNewNumber(long data, long factor) {
        long ans = 0;

月饼是中国人在中秋佳节时吃的一种传统食品，不同地区有许多不同风味的月饼。现给定所有种类月饼的库存量、总售价、以及市场的最大需求量，请你计算可以获得的最大收益是多少。
```
import org.omg.PortableInterceptor.INACTIVE;

import java.lang.reflect.Array;
import java.util.*;
import java.io.*;
import java.math.*;
public class Main{

    public static void main(String[] args) throws Exception{
        Scanner scanner = new Scanner(System.in);
        int size = scanner.nextInt();
        int allOfNeed = scanner.nextInt();
        Vector<Cake> vector = new Vector<>();
        for(int i=0; i<size; i++) {
            Cake cake = new Cake();
            cake.setAmount(scanner.nextInt());
            vector.add(cake);
        }
        for(int i=0; i<size; i++) {
            Cake cake = vector.get(i);
            cake.setPrice(scanner.nextInt());
            cake.setFactor((1.0)*cake.getPrice()/cake.getAmount());
        }
        Collections.sort(vector, new Comparator<Cake>() {
            @Override
            public int compare(Cake o1, Cake o2) {
                if(o2.getFactor()>o1.getFactor())
                    return 1;
                else if(o2.getFactor()-o1.getFactor()==0)
                    return 0;
                else
                    return -1;
            }
        });

        double sum = 0;
        for(int i=0; i<vector.size(); i++) {
//            System.out.println(vector.get(i).getAmount()+"-"+vector.get(i).getPrice()+"-"+vector.get(i).getFactor());
            if(vector.get(i).getAmount()<=allOfNeed) {
                sum += vector.get(i).getPrice();
                allOfNeed -= vector.get(i).getAmount();
            }
            else {
                sum += vector.get(i).getFactor()*allOfNeed;
                allOfNeed = 0;
            }
            if(allOfNeed==0)
                break;
        }
        System.out.println(new java.text.DecimalFormat("#.00").format(sum));
    }
}

        Long tmp = data;
        Stack<Character> stack = new Stack<Character>();
        StringBuilder result = new StringBuilder(0);
        while (tmp != 0) {
            stack.add(array[new Long((tmp % N)).intValue()]);
            tmp = tmp / N;
        }
        for (; !stack.isEmpty();) {
            result.append(stack.pop());
        }
        return result.length() == 0 ? "0":result.toString();
    }
    // 任意进制转10进制
    public static long Hexadecimal_From_Random_To_10(String number, int N) {

        if (N == 10){
            return Long.parseLong(number);
        }
        long base = 1;
        for (int i = len - 1; i >= 0; i--) {
            int index = numStr.indexOf(ch[i]);
            result += index * base;
            base *= N;
        }
        return result;

import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        while (in.hasNext()) {
            String A = in.next();
            String B = in.next();
            Map<Character, Integer> map = new HashMap<>();
            for(int i=0; i<A.length(); i++) {
                map.put(A.charAt(i), map.getOrDefault(A.charAt(i), 0)+1);
            }
            boolean flag = true;
            for(int i=0; i<B.length(); i++) {
                // 没找到
                if(map.getOrDefault(B.charAt(i), 0)==0) {
                    flag = false;
                    break;
                }
                // 减1
                map.put(B.charAt(i), map.getOrDefault(B.charAt(i), 0)-1);

            }
            if(flag==true)
                System.out.println("Yes");
            else
                System.out.println("No");
        }
    }
}
```

## 蟠桃记

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String[] data = new String[91];
        data[1] = "1";
        for(int i=2; i<=30; i++) {
            data[i] = (new BigInteger(data[i-1]).add(new BigInteger("1"))).multiply(new BigInteger("2")).toString();
        }
        while (scanner.hasNext()) {
            int index = scanner.nextInt();
            if(index==0)
                break;
            if(index==1) {
                System.out.println(1);
                continue;
            }
            System.out.println(data[index]);
        }
    }
}

## 成绩排名

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int size = scanner.nextInt();
        Map<Integer, String> map = new HashMap<>();
        int max = Integer.MIN_VALUE;
        int min = Integer.MAX_VALUE;
        while (size-->0) {
            String name = scanner.next();
            String classroom = scanner.next();
            int point = scanner.nextInt();
            map.put(point, name + " " + classroom);
            if(point > max)
                max = point;
            if(point < min)
                min = point;
        }
        System.out.println(map.get(max));
        System.out.println(map.get(min));
    }
}
```

## 分解因数

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        while (scanner.hasNext()) {
            int data = scanner.nextInt();
            System.out.print(data + " =");
            for(int i=2; i<=Math.sqrt(data); i++) {
                while (data!=i) {
                    if(data%i==0) {
                        System.out.print(" " + i + " *");
                        data /= i;
                    }
                    else
                        break;
                }
            }
            System.out.println(" " + data);
        }

    }
}

```

## 斐波那契凤尾

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int[] data = new int[100001];
        data[0] = 1;
        data[1] = 1;
        for(int i=2; i<100001; i++) {
            data[i] = (data[i-1] + data[i-2]) % 1000000;
        }
        while (scanner.hasNext()) {
            int index = scanner.nextInt();
            System.out.printf((index<25?"%d\n":"%06d\n"), data[index]);
        }
    }
}

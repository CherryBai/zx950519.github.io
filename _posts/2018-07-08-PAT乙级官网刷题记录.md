## 1001 害死人不偿命的(3n+1)猜想 (15)（15 分）

![](https://ws1.sinaimg.cn/large/005L0VzSgy1ft2sbqbmfkj30l30ae752.jpg)  

```
import java.util.*;
public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        while (scanner.hasNext()) {
            int data = scanner.nextInt();
            int res = 0;
            while (data != 1) {
                if(data%2==0) {
                    data /= 2;
                }
                else
                    data = (data * 3 + 1) / 2;
                res++;
            }
            System.out.println(res);
        }
    }
}
```

```
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Map<Character, String> map = new HashMap<>();
        map.put('0', "ling");map.put('1', "yi");map.put('2', "er");map.put('3', "san");map.put('4', "si");
        map.put('5', "wu");map.put('6', "liu");map.put('7', "qi");map.put('8', "ba");map.put('9', "jiu");
        while (scanner.hasNext()) {
             String data = scanner.next();
             double tmp = 0;
             for(int i=0; i<data.length(); i++) {
                 tmp += Double.parseDouble(data.substring(i, i+1));
             }
             String res = String.valueOf((int)(tmp));
             for(int i=0; i<res.length(); i++) {
                if(i!=0)
                    System.out.print(" ");
                System.out.print(map.get(res.charAt(i)));
            }
            System.out.println();
        }
    }
}
```

## 1007 素数对猜想 (20)（20 分）
![](https://ws1.sinaimg.cn/large/005L0VzSgy1ft54joyurfj30qb09374s.jpg)  

```
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int size = scanner.nextInt();
        boolean[] data = new boolean[size+1];
        for(int i=2; i<=size/2; i++) {
            int factor = 2;
            while (true) {
                if(i*factor<=size) {
                    data[i*factor] = true;
                }
                else
                    break;
                factor++;
            }
        }
        Vector<Integer> vector = new Vector<>();
        for(int i=2; i<=size; i++) {
            if(data[i]==false)
                vector.add(i);
        }
        int sum = 0;
        for(int i=1; i<vector.size(); i++) {
            if(vector.get(i)-vector.get(i-1)==2)
                sum++;
        }
        System.out.println(sum);
    }
}
```

## 1008 数组元素循环右移问题 (20)（20 分）
![](https://ws1.sinaimg.cn/large/005L0VzSgy1ft7fm203cdj315c08nq3w.jpg)  

```
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int size = scanner.nextInt();
        int p = scanner.nextInt();
        int[] data = new int[size];
        for(int i=0; i<size; i++) {
            data[(i+p)%size] = scanner.nextInt();
        }
        for(int i=0; i<data.length; i++) {
            if(i!=0)
                System.out.print(" ");
            System.out.print(data[i]);
        }
        System.out.println();
    }
}
```

## 1011 A+B和C (15)（15 分）
![](https://ws1.sinaimg.cn/large/005L0VzSgy1ft54waynghj30qg0cijs5.jpg)  

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int size = scanner.nextInt();
        int tmp = size;
        while (size-->0) {
            BigInteger a = new BigInteger(scanner.next());
            BigInteger b = new BigInteger(scanner.next());
            BigInteger c = new BigInteger(scanner.next());
            int flag = a.add(b).subtract(c).compareTo(new BigInteger("0"));
            System.out.print("Case #");
            System.out.print(tmp-size + ": ");
            if(flag > 0)
                System.out.println("true");
            else
                System.out.println("false");
        }
    }
}
```

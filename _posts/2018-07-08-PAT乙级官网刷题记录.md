```
        boolean[] data = new boolean[size+1];

            int factor = 2;
            while (true) {
                if(i*factor<=size) {
                    data[i*factor] = true;

                else
```

```
    public static String Hexadecimal_From_10_To_Random(long data, int N) {
        Long tmp = data;
        Stack<Character> stack = new Stack<Character>();
        StringBuilder result = new StringBuilder(0);
        while (tmp != 0) {
            stack.add(array[new Long((tmp%N)).intValue()]);
            tmp = tmp / N;
        }
        for(; !stack.empty(); ) {
            result.append(stack.pop());
        }
        return result.length()==0 ? "0": result.toString();
    }
```

```
        for(int i=0; i<data.length; i++) {

            map.put(i, Integer.parseInt(data[i]));
        }
        StringBuilder res = new StringBuilder("");
        for(Map.Entry<Integer, Integer> entry: map.entrySet()) {
            if(entry.getKey()==0)
                continue;
            if(entry.getValue()!=0) {
                res.append(entry.getKey());
                map.put(entry.getKey(), entry.getValue()-1);
                break;
            }
        }
        for(Map.Entry<Integer, Integer> entry: map.entrySet()) {
            while (entry.getValue()>0) {
                res.append(entry.getKey());
                map.put(entry.getKey(), entry.getValue()-1);
            }
        }
        System.out.println(res);
```

```
                map.put(data.charAt(i),
                    map.getOrDefault(data.charAt(i), 0)+1);
        }
        int max = Integer.MIN_VALUE;

        }
```

## 1054 求平均值 (20)（20 分）  

![](https://ws1.sinaimg.cn/large/005L0VzSgy1ftqvewvrhvj30mr05zwf3.jpg)  

```
import java.math.BigDecimal;
import java.math.BigInteger;
import java.math.RoundingMode;
import java.text.DecimalFormat;
import java.util.*;
import java.util.regex.Pattern;

public class Main {


    public static void main(String[] args) {

       Scanner scanner = new Scanner(System.in);
       int size = Integer.parseInt(scanner.nextLine());

       String datas = scanner.nextLine();
       String[] data = datas.split(" ");

       double sum = 0;
       double ava_num = 0;
       for(int i=0; i<size; i++) {
           try {
               double tmp = Double.parseDouble(data[i]);
               int time = 0;
               if(data[i].contains("."))
                   time = data[i].length() - data[i].indexOf(".") - 1;
               if(Math.abs(tmp) <= 1000 && Math.abs(time) <= 2) {
                   sum += tmp;
                   ava_num++;
               }
               else {
                   throw new Exception();
               }
           }
           catch (Exception e) {
               System.out.printf("ERROR: %s is not a legal number\n", data[i]);
           }
       }
        if (ava_num == 0) {
            System.out.printf("The average of %.0f numbers is Undefined", ava_num);
        } else if (ava_num == 1) {
            System.out.printf("The average of %.0f number is %.2f\n", ava_num, sum / ava_num);
        } else {
            System.out.printf("The average of %.0f numbers is %.2f\n", ava_num, sum / ava_num);
        }

    }

}
```
